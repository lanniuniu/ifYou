- table
‌主键（Primary Key）‌：唯一标识表中每一行数据的列，不能为空且不能重复（通常是 id）。
‌外键（Foreign Key）‌：用于建立表与表之间关联的列，指向另一张表的主键。
‌索引（Index）‌：类似于书籍的目录，用于加速数据查询速度。
```
// 创建表
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY AUTO_INCREMENT, // 自增主键
    username VARCHAR(255) NOT NULL, // 非空
    email VARCHAR(255) UNIQUE, // 唯一 可以为空
    age INT
)

// 插入数据
INSERT INTO users (username, email, age) 
VALUES ('alice', 'alice@example.com', 25);

// 查询
SELECT * FROM users WHERE age > 20;

-- 添加新列
ALTER TABLE users ADD COLUMN phone VARCHAR(20);

-- 修改列类型
ALTER TABLE users MODIFY COLUMN age SMALLINT;

// 删除表
DROP TABLE users;

// 注释
COMMENT ""
```
